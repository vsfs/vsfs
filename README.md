# Dependancies

 - automake/autoconf/libtool
 - g++ > 4.7.1
 - boost >= 1.53
 - thrift 0.9.0
 - FUSE > 2.8.0
 - LevelDB and snappy
 - glog
 - gflags
 - libssl-dev
 - libattr1-dev
 - gmock 1.6.0 for testing.
